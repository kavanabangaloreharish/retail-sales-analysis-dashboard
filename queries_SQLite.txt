 -- Query 1: Top 10 products by total revenue
SELECT 
    "Product Name",
    ROUND(SUM(Sales), 2) AS total_sales
FROM superstore
GROUP BY "Product Name"
ORDER BY total_sales DESC
LIMIT 10;

 -- Query 2: Top 10 Customers by Revenue
SELECT 
    "Customer Name",
    ROUND(SUM(Sales), 2) AS total_sales,
    ROUND(SUM(Profit), 2) AS total_profit
FROM superstore
GROUP BY "Customer Name"
ORDER BY total_sales DESC
LIMIT 10;

 -- Query 3: Worst-Performing Categories (by Profit)
SELECT 
    Category,
    "Sub-Category",
    ROUND(SUM(Sales), 2) AS total_sales,
    ROUND(SUM(Profit), 2) AS total_profit
FROM superstore
GROUP BY Category, "Sub-Category"
ORDER BY total_profit ASC
LIMIT 10;

-- Query 4: Revenue Trend (Monthly, using CTE + window function)
WITH monthly_sales AS (
    SELECT 
        strftime('%Y-%m', Date("Order Date")) AS month,
        SUM(Sales) AS total_sales
    FROM superstore
    GROUP BY month
)
SELECT 
    month,
    ROUND(total_sales, 2) AS total_sales,
    ROUND(total_sales - LAG(total_sales) OVER (ORDER BY month), 2) AS sales_change
FROM monthly_sales
ORDER BY month;

-- Note: This dataset is a single denormalized table, so no joins were 
-- needed for this analysis. Joins would apply if Customers, Orders, 
-- and Products were separate tables.
